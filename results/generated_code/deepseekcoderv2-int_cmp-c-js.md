### Solution
```
let a = parseInt(prompt("Enter the first integer:"));
let b = parseInt(prompt("Enter the second integer:"));
if (a < b) {
  console.log(`${a} is less than ${b}`);
} else if (a === b) {
  console.log(`${a} is equal to ${b}`);
} else {
  console.log(`${a} is greater than ${b}`);
}
```
