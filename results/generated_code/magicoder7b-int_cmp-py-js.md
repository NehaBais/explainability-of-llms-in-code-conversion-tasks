 ```javascript
// Get two integers from the user.
let a = prompt('Enter value of a');
let b = prompt('Enter value of b');

if (a < b) {
    // Display a message if the first integer is less than the second one.
    console.log('a is less than b');
} else if (a > b) {
    // Display a message if the first integer is greater than the second one.
    console.log('a is greater than b');
} else if (a == b) {
    // Display a message if the first integer is equal to the second one.
    console.log('a is equal to b');
}
```
