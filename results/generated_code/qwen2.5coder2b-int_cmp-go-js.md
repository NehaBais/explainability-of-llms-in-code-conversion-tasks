```javascript
// Get two integers from the user.
const num1 = prompt("Enter number: ");
const num2 = prompt("Enter number: ");

// Display a message if the first integer is less than, equal to or greater than the second integer.
if(num1 < num2){
  console.log(`${num1} less than ${num2}`);
}else if(num1 == num2){
  console.log(`${num1} equal to ${num2}`);
}else{
  console.log(`${num1} greater than ${num2}`);
}
```
