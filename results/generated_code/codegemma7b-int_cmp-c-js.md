```javascript
const a = parseInt(prompt("Enter the first integer:"));
const b = parseInt(prompt("Enter the second integer:"));

if (a < b) {
  console.log(`${a} is less than ${b}`);
}

if (a === b) {
  console.log(`${a} is equal to ${b}`);
}

if (a > b) {
  console.log(`${a} is greater than ${b}`);
}
```
